### 二次封装
src-api-request.js
```js
import axios from 'axios'

const request = axios.create({
    baseURL:'/api',
    timeout:5000
})
//拦截器
request.interceptors.request.use((config)=>{
    return config
})

request.interceptors.response.use((res)=>{
    return res.data
},(err)=>{
    return Promise.reject(new Error('faile'))
})

export default request;
```
